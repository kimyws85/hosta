C++이란? (C와의 차이)

C++ : 1983년 비야네 스트롭스트룹(Bjarne Stroustrup)이 C에 객체지향 패러다임을 더해 만든 언어. "C with Classes"에서 출발.

하위 호환 : C 코드를 대부분 그대로 사용 가능. C의 저수준 제어 능력을 유지하면서 고수준 추상화를 추가.

멀티 패러다임 : 절차지향 + 객체지향 + 제네릭 + 함수형(일부) 프로그래밍을 모두 지원.

표준 : C++11, C++14, C++17, C++20, C++23 — 버전마다 새 기능이 추가된다.

연산자(Operator) : 피연산자에 특정 연산을 수행하는 기호. C++에는 약 50가지 이상의 연산자가 있다.

우선순위 함정 예제
#include <iostream>
using namespace std;

int main() {
    int a = 2, b = 3, c = 4;

    // * 이 + 보다 우선순위 높음
    cout << a + b * c << "\n";    // 14  (2 + 12)
    cout << (a + b) * c << "\n";  // 20  (5 × 4)

    // 비교 < 이 비트 & 보다 우선순위 높음!
    int x = 4, y = 5;
    cout << (x & y == 4) << "\n";   // 0 — y==4 먼저, 5==4→0, 4&0=0
    cout << ((x & y) == 4) << "\n"; // 1 — 4&5=4, 4==4→true

    // 대입은 우→좌 결합
    int p, q, r;
    p = q = r = 10;   // r=10, q=10, p=10 순서
    cout << p << " " << q << " " << r << "\n";  // 10 10 10

    // 후위 vs 전위 증감
    int n = 5;
    cout << n++ << "\n";  // 5  (출력 후 증가)
    cout << n   << "\n";  // 6
    cout << ++n << "\n";  // 7  (증가 후 출력)
}

산술 연산자 (+, -, *, /, %, 증감)

정수 나눗셈 : int / int는 소수점 버림. 실수 나눗셈을 원하면 형변환 필요.

나머지(%) : 정수 전용. 음수에서 결과 부호는 피제수(왼쪽)를 따름.

오버플로 : int는 약 ±21억. 넘으면 undefined behavior (서명 정수) 또는 wrap-around (무부호).

산술 연산자 완전 예제
#include <iostream>
#include <cmath>    // pow, sqrt, fmod
using namespace std;

int main() {
    // 기본 산술
    int a = 17, b = 5;
    cout << a + b << "\n";  // 22
    cout << a - b << "\n";  // 12
    cout << a * b << "\n";  // 85
    cout << a / b << "\n";  // 3  (정수 나눗셈, 소수 버림)
    cout << a % b << "\n";  // 2  (나머지)

    // 실수 나눗셈 (형변환 필요)
    cout << (double)a / b << "\n";    // 3.4
    cout << (double)a / b * 100 << "\n"; // 340 (백분율 계산)

    // 음수 나머지
    cout << -7 % 3  << "\n";  // -1 (피제수 부호 따름)
    cout <<  7 % -3 << "\n";  // 1

    // cmath 활용
    cout << pow(2, 10) << "\n";    // 1024
    cout << sqrt(144) << "\n";    // 12
    cout << fmod(7.5, 2.5) << "\n"; // 2.5 (실수 나머지)

    // 오버플로 확인
    int      si = 2147483647;   // INT_MAX
    unsigned ui = 4294967295u;  // UINT_MAX
    cout << si + 1 << "\n";  // UB (signed overflow)
    cout << ui + 1 << "\n";  // 0  (unsigned wrap)
}

using namespace std;

    int x = i++;    // x=3, i=4

    int y = ++i;    // i=5, y=5

    int arr[] = {10, 20, 30, 40, 50};
    int sum = 0;
    for (int k = 0; k < 5; ++k)  // 전위 권장
        sum += arr[k];
    cout << "sum=" << sum << "\n";
}

비교 연산자 : == != < <= > >= — 결과는 항상 bool.

비교 연산자 + 부동소수점 함정
#include <iostream>
#include <cmath>
#include <limits>
using namespace std;

bool almostEqual(double a, double b) {
    return fabs(a - b) < numeric_limits<double>::epsilon() * 100;
}

int main() {
    // 정수 비교
    int a = 10, b = 20;
    cout << boolalpha;            // true/false로 출력
    cout << (a == b) << "\n";   // false
    cout << (a != b) << "\n";   // true
    cout << (a <= b) << "\n";   // true

    // 부동소수점 비교 함정
    double x = 0.1 + 0.2;
    cout << (x == 0.3) << "\n";           // false!
    cout << almostEqual(x, 0.3) << "\n"; // true (엡실론 비교)

    // 문자 비교 (ASCII 순서)
    cout << ('a' < 'z') << "\n";  // true  (97 < 122)
    cout << ('A' < 'a') << "\n";  // true  (65 < 97)

    // 문자열 비교 (사전순)
    string s1 = "apple", s2 = "banana";
    cout << (s1 < s2) << "\n";  // true
    cout << (s1 == "apple") << "\n"; // true
}

우주선 연산자 <=> (C++20)
#include <iostream>
#include <compare>
#include <vector>
#include <algorithm>
using namespace std;

struct Version {
    int major_, minor_, patch_;

    // <=> 하나만 정의하면 ==, !=, <, <=, >, >= 모두 자동 생성
    auto operator<=>(const Version&) const = default;
    bool operator==(const Version&) const = default;
};

int main() {
    Version v1{2, 1, 0};
    Version v2{2, 3, 1};
    Version v3{1, 9, 9};

    cout << boolalpha;
    cout << (v1 < v2) << "\n";   // true
    cout << (v3 < v1) << "\n";   // true

    vector<Version> versions = {v2, v1, v3};
    sort(versions.begin(), versions.end());
    for (auto& v : versions)
        cout << v.major_ << "." << v.minor_ << "." << v.patch_ << "\n";
}

논리 연산자 & 단락 평가 (Short-Circuit)

&& (AND) : 왼쪽이 false면 오른쪽 평가 안 함.

|| (OR) : 왼쪽이 true면 오른쪽 평가 안 함.

! (NOT) : bool 반전. !!x로 임의 타입을 bool로 변환.

단락 평가 활용 : 포인터 null 체크, 인덱스 범위 체크 등 안전한 패턴에 필수.

단락 평가 (Short-Circuit Evaluation)
#include <iostream>
#include <vector>
using namespace std;

bool check(const string& msg) {
    cout << "check(" << msg << ") 호출\n";
    return msg == "ok";
}

int main() {
    // && : 왼쪽 false → 오른쪽 호출 안 됨
    cout << "=== && 테스트 ===\n";
    bool r1 = check("fail") && check("ok");
    cout << "결과: " << boolalpha << r1 << "\n\n";

    // || : 왼쪽 true → 오른쪽 호출 안 됨
    cout << "=== || 테스트 ===\n";
    bool r2 = check("ok") || check("ok");
    cout << "결과: " << r2 << "\n\n";

    // 실전: null 포인터 & 배열 범위 안전 체크
    vector<int> v = {1, 2, 3};
    int idx = 5;
    if (idx < (int)v.size() && v[idx] > 0)  // 안전: 범위 먼저 체크
        cout << "접근 가능\n";
    else
        cout << "범위 초과 — v[5] 접근 안 함\n";

    // null 체크 패턴
    int* ptr = nullptr;
    if (ptr != nullptr && *ptr > 0)  // null 체크 먼저
        cout << *ptr;
    else
        cout << "nullptr 안전 처리\n";

    // !! 패턴: 임의 값을 bool로
    int n = 42;
    bool flag = !!n;  // true (0이 아닌 값)
    cout << "!!42 = " << flag << "\n";
}

비트 연산자 — 비트마스킹 & 최적화

비트 연산자 : & | ^ ~ << >> — 정수를 비트 단위로 조작.

활용 : 플래그 관리, 권한 시스템, 암호화, 압축, 고속 산술(시프트 = ×2, ÷2).

주의 : 부호 있는 정수의 오른쪽 시프트(>>)는 구현 정의(보통 산술 시프트). 비트 연산은 unsigned 타입 권장.

#include <iostream>
#include <bitset>
using namespace std;

// 파일 권한 플래그
constexpr uint8_t READ    = 1 << 0;  // 0000 0001 = 1
constexpr uint8_t WRITE   = 1 << 1;  // 0000 0010 = 2
constexpr uint8_t EXECUTE = 1 << 2;  // 0000 0100 = 4
constexpr uint8_t HIDDEN  = 1 << 3;  // 0000 1000 = 8

void printPerms(uint8_t p) {
    cout << "비트: " << bitset<8>(p) << "  ";
    cout << (p & READ    ? "r" : "-");
    cout << (p & WRITE   ? "w" : "-");
    cout << (p & EXECUTE ? "x" : "-");
    cout << (p & HIDDEN  ? "h" : "-");
    cout << "\n";
}

int main() {
    uint8_t perm = READ | WRITE;        // rw--
    printPerms(perm);

    perm |= EXECUTE;                    // 실행 권한 추가
    printPerms(perm);

    perm &= ~WRITE;                     // 쓰기 권한 제거
    printPerms(perm);

    perm ^= HIDDEN;                     // 숨김 토글 ON
    printPerms(perm);

    // 시프트로 빠른 2의 거듭제곱
    cout << "--- 시프트 ×/÷ 2 ---\n";
    for (int i = 0; i < 8; i++)
        cout << "1 << " << i << " = " << (1 << i) << "\n";
}

#include <iostream>
using namespace std;

int main() {
    // XOR 스왑 (임시 변수 없이)
    int a = 42, b = 17;
    a ^= b; b ^= a; a ^= b;
    cout << "a=" << a << " b=" << b << "\n";  // a=17 b=42

    // 홀짝 판별: 최하위 비트
    for (int n = 0; n <= 5; n++)
        cout << n << (n & 1 ? " 홀수" : " 짝수") << "\n";

    // 2의 거듭제곱 체크: n & (n-1) == 0
    auto isPow2 = [](int n) {
        return n > 0 && (n & (n - 1)) == 0;
    };
    for (int n : {1, 2, 3, 4, 7, 8, 16, 100})
        cout << n << ": " << (isPow2(n) ? "2의 거듭제곱" : "아님") << "\n";

    // 비트 카운트 (켜진 비트 수)
    auto popcount = [](unsigned n) {
        int cnt = 0;
        while (n) { cnt += n & 1; n >>= 1; }
        return cnt;
    };
    cout << "255의 비트 수: " << popcount(255) << "\n";  // 8
    cout << "170의 비트 수: " << popcount(170) << "\n";  // 4 (10101010)
}

sizeof / typeid / decltype / alignof

sizeof : 타입·변수의 바이트 크기를 컴파일 타임에 반환. 런타임 비용 0.

typeid : 런타임 타입 정보(RTTI). <typeinfo> 헤더 필요. 다형성 클래스에서 유용.

decltype (C++11) : 표현식의 타입을 추론. 제네릭 코드에서 반환 타입 지정에 사용.

alignof (C++11) : 타입의 정렬 요구사항(바이트)을 반환.

sizeof / typeid / decltype
#include <iostream>
#include <typeinfo>
#include <vector>
using namespace std;

struct Pair { char a; int b; };       // 패딩 포함 → 8바이트
struct Packed { char a; char b; };    // 2바이트

template<typename A, typename B>
auto add(A a, B b) -> decltype(a + b) {
    return a + b;
}

int main() {
    // sizeof 기본 타입
    cout << "sizeof(char)   = " << sizeof(char)   << "\n";
    cout << "sizeof(short)  = " << sizeof(short)  << "\n";
    cout << "sizeof(int)    = " << sizeof(int)    << "\n";
    cout << "sizeof(long)   = " << sizeof(long)   << "\n";
    cout << "sizeof(double) = " << sizeof(double) << "\n";
    cout << "sizeof(Pair)   = " << sizeof(Pair)   << "\n"; // 8 (패딩)
    cout << "sizeof(Packed) = " << sizeof(Packed) << "\n"; // 2

    // 배열 길이 계산
    int arr[] = {1, 2, 3, 4, 5};
    int len = sizeof(arr) / sizeof(arr[0]);
    cout << "배열 길이: " << len << "\n";

    // typeid
    double d = 3.14;
    cout << "타입: " << typeid(d).name() << "\n";

    // decltype: 반환 타입 자동 추론
    auto r1 = add(1, 2);       // int
    auto r2 = add(1, 2.5);     // double
    cout << typeid(r1).name() << "\n";  // i (int)
    cout << typeid(r2).name() << "\n";  // d (double)

    // alignof
    cout << "alignof(int)    = " << alignof(int)    << "\n";
    cout << "alignof(double) = " << alignof(double) << "\n";
}

형변환 연산자 — C 스타일 vs C++ 캐스트

C 스타일 (int)x : 강력하지만 위험. 컴파일러가 의도를 파악 못함.

static_cast : 컴파일 타임 타입 변환. 안전한 명시적 변환 (int↔double, enum↔int).

dynamic_cast : 런타임 다운캐스트. RTTI 필요. 실패 시 nullptr 반환(포인터) or bad_cast(참조).

const_cast : const 속성 추가/제거. 오래된 API 대응에만 사용.

reinterpret_cast : 비트 수준 재해석. 매우 위험 — 네트워크/임베디드 저수준에만.

dynamic_cast런타임다형성 다운캐스트높음 (검사됨)

#include <iostream>
#include <cmath>
using namespace std;

class Vec2 {
public:
    double x, y;
    Vec2(double x=0, double y=0) : x(x), y(y) {}

    // 덧셈/뺄셈
    Vec2 operator+(const Vec2& o) const { return {x+o.x, y+o.y}; }
    Vec2 operator-(const Vec2& o) const { return {x-o.x, y-o.y}; }
    Vec2& operator+=(const Vec2& o) { x+=o.x; y+=o.y; return *this; }

    // 스칼라 곱
    Vec2 operator*(double s) const { return {x*s, y*s}; }
    friend Vec2 operator*(double s, const Vec2& v) { return v * s; }

    // 내적 (dot product)
    double operator*(const Vec2& o) const { return x*o.x + y*o.y; }

    // 크기
    double length() const { return sqrt(x*x + y*y); }

    // 단항 마이너스
    Vec2 operator-() const { return {-x, -y}; }

    // 비교
    bool operator==(const Vec2& o) const { return x==o.x && y==o.y; }
    bool operator!=(const Vec2& o) const { return !(*this==o); }

    // 출력
    friend ostream& operator<<(ostream& os, const Vec2& v) {
        return os << "(" << v.x << ", " << v.y << ")";
    }
    // 인덱스
    double& operator[](int i) { return i == 0 ? x : y; }
};

int main() {
    Vec2 a{3, 4}, b{1, 2};

    cout << "a = "     << a << "\n";
    cout << "b = "     << b << "\n";
    cout << "a+b = "   << a+b << "\n";
    cout << "a-b = "   << a-b << "\n";
    cout << "a*2 = "   << a*2 << "\n";
    cout << "3*b = "   << 3*b << "\n";

}

스택(Stack) 클래스 — 연산자 오버로딩
#include <iostream>
#include <vector>
#include <stdexcept>
using namespace std;

template<typename T>
class Stack {
    vector<T> data_;
public:
    Stack& operator<<(const T& val) {  // push: stack << val
        data_.push_back(val); return *this;
    }
    T operator()() {                    // pop: stack()
        if (data_.empty()) throw underflow_error("empty stack");
        T v = data_.back(); data_.pop_back(); return v;
    }
    T& operator*() {                    // top: *stack
        if (data_.empty()) throw underflow_error("empty stack");
        return data_.back();
    }
    explicit operator bool() const {  // bool 변환: if(stack)
        return !data_.empty();
    }
    int operator!() const {           // !stack → size
        return (int)data_.size();
    }
};

int main() {
    Stack<int> s;
    s << 10 << 20 << 30;   // push 연쇄
    cout << "size: " << !s << "\n";       // 3
    cout << "top:  " << *s << "\n";       // 30
    while (s)
        cout << s() << " ";              // 30 20 10
    cout << "\n";
    cout << "empty: " << boolalpha << !(bool)s << "\n";

#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() {
    // 정수 조건 (0 = false, 나머지 = true)
    int n = 42;
    if (n)       cout << "n은 0이 아님\n";
    if (n % 2)   cout << "홀수\n";
    else         cout << "짝수\n";

    // 포인터 조건 (nullptr = false)
    int* p = &n;
    if (p)       cout << "유효한 포인터: " << *p << "\n";
    p = nullptr;
    if (!p)      cout << "nullptr\n";

    // 문자열 비어있는지
    string s = "hello";
    if (!s.empty()) cout << "문자열: " << s << "\n";

    // 벡터 크기 조건
    vector<int> v = {1, 2, 3};
    if (v.size() > 2 && v.back() > 0)
        cout << "벡터 조건 충족\n";

    // dangling else 주의
    int x = 5, y = 3;
    if (x > 0)
        if (y > 10)
            cout << "y도 크다\n";
        else      // ← 이 else는 if(x>0)이 아닌 if(y>10)에 붙음!
            cout << "y는 작다 (x>0 이지만)\n";
}

enum class Direction { North, South, East, West };
enum class Terrain   { Plain, Mountain, River, Forest };

struct Player { int x = 0, y = 0; };

int moveCost(Terrain t) {
    switch (t) {
        case Terrain::Plain:    return 1;
        case Terrain::Forest:   return 2;
        case Terrain::Mountain: return 4;
        case Terrain::River:    return 3;
    }
    return 1;
}

void move(Player& p, Direction d, int steps = 1) {
    switch (d) {
        case Direction::North: p.y += steps; break;
        case Direction::South: p.y -= steps; break;
        case Direction::East:  p.x += steps; break;
        case Direction::West:  p.x -= steps; break;
    }
}

string dirName(Direction d) {
    switch (d) {
        case Direction::North: return "북";
        case Direction::South: return "남";
        case Direction::East:  return "동";
        case Direction::West:  return "서";
    }
    return "?";
}

int main() {
    Player hero;
    vector<Direction> path = {
        Direction::North, Direction::North,
        Direction::East,  Direction::South
    };

    int totalCost = 0;
    for (auto d : path) {
        move(hero, d);
        Terrain t = Terrain::Plain;  // 실제론 맵 조회
        totalCost += moveCost(t);
        cout << dirName(d) << "으로 이동 → ("
             << hero.x << "," << hero.y << ")\n";
    }
    cout << "총 이동 비용: " << totalCost << "\n";
}

조건문 성능 — 분기 예측 & 최적화 패턴

분기 예측(Branch Prediction) : CPU는 if 결과를 미리 예측해 실행. 예측 실패 시 파이프라인 플러시 → 수십 사이클 손실.

[[likely]] / [[unlikely]] (C++20) : 컴파일러에 분기 확률 힌트 제공. 핫 경로를 빠르게.

최적화 패턴 : 자주 참인 조건을 앞에, 빠른 조건을 먼저, switch가 if-else보다 점프 테이블로 빠를 수 있음.

[[likely]] / [[unlikely]] & 분기 최소화
#include <iostream>
#include <vector>
using namespace std;

// C++20 [[likely]] / [[unlikely]]
int classify(int n) {
    if (n > 0) [[likely]] {     // 대부분의 케이스
        return 1;
    } else if (n == 0) [[unlikely]] {
        return 0;
    } else [[unlikely]] {
        return -1;
    }
}

// 분기 없이 조건 처리 (branchless)
int absVal(int x) {
    int mask = x >> 31;      // 음수면 -1(0xFFFFFFFF), 양수면 0
    return (x ^ mask) - mask; // 분기 없는 절대값
}

int maxVal(int a, int b) {
    return a + ((b - a) & -((b - a) >> 31)); // 분기 없는 max
}

int main() {
    cout << classify(5)  << "\n";   // 1
    cout << classify(0)  << "\n";   // 0
    cout << classify(-3) << "\n";   // -1

    cout << absVal(-42) << "\n";    // 42 (분기 없음)
    cout << maxVal(7, 3) << "\n";  // 7 (분기 없음)

    // 조건부 표현식을 배열 인덱스로
    const char* msg[] = {"짝수", "홀수"};
    for (int i = 0; i < 6; i++)
        cout << i << ": " << msg[i & 1] << "\n"; // 분기 없음
}

상태 머신 (State Machine) — 조건문의 실전 응용

유한 상태 머신(FSM) : 현재 상태 + 입력 → 다음 상태 + 출력. switch + enum class가 자연스럽게 맞음.

활용 : 게임 AI, 프로토콜 파서, 텍스트 처리, 주문/결제 흐름 등 거의 모든 도메인.

#include <iostream>
using namespace std;

enum class State { Idle, HasMoney, Dispensing };
enum class Event { InsertCoin, SelectItem, Dispense, Cancel };

// 절차적 스타일: 상태·잔액을 변수로 관리, 전이를 함수로 처리
void handleEvent(State& state, int& balance, Event ev, int amount) {
    const int PRICE = 1000;

    switch (state) {
        case State::Idle:
            if (ev == Event::InsertCoin) {
                balance += amount;
                state = State::HasMoney;
                cout << amount << "원 투입. 잔액: " << balance << "\n";
            }
            break;

        case State::HasMoney:
            if (ev == Event::InsertCoin) {
                balance += amount;
                cout << amount << "원 추가. 잔액: " << balance << "\n";
            } else if (ev == Event::SelectItem) {
                if (balance >= PRICE) {
                    state = State::Dispensing;
                    cout << "상품 선택됨. 배출 중...\n";
                } else {
                    cout << "잔액 부족. " << PRICE - balance << "원 더 필요\n";
                }
            } else if (ev == Event::Cancel) {
                cout << balance << "원 반환\n";
                balance = 0;
                state = State::Idle;
            }
            break;

        case State::Dispensing:
            if (ev == Event::Dispense) {
                cout << "상품 배출! 거스름돈: " << balance - PRICE << "원\n";
                balance = 0;
                state = State::Idle;
            }
            break;
    }
}

int main() {
    State state   = State::Idle;
    int   balance  = 0;

    handleEvent(state, balance, Event::InsertCoin,  500);
    handleEvent(state, balance, Event::SelectItem,  0);   // 잔액 부족
    handleEvent(state, balance, Event::InsertCoin,  700);
    handleEvent(state, balance, Event::SelectItem,  0);
    handleEvent(state, balance, Event::Dispense,    0);
}

// 변수 저장소 (절차적 스타일)
map<string, int> vars;

int getVar(const string& name) {
    if (auto it = vars.find(name); it != vars.end())
        return it->second;
    return 0;
}

void runLine(const string& cmd) {
    if (cmd.empty() || cmd[0] == '#') return;
    istringstream line(cmd);
    string kw;
    line >> kw;

    if (kw == "let") {
        string var, eq; int val;
        line >> var >> eq >> val;
        vars[var] = val;
    } else if (kw == "print") {
        string var; line >> var;
        cout << var << " = " << getVar(var) << "\n";
    } else if (kw == "add") {
        string r, a, b; line >> r >> a >> b;
        vars[r] = getVar(a) + getVar(b);
    } else if (kw == "ifgt") {
        string a, b, msg; line >> a >> b;
        getline(line, msg);
        if (getVar(a) > getVar(b))
            cout << "[if]" << msg << "\n";
    } else {
        cout << "[오류] 모르는 명령: " << kw << "\n";
    }
}

void runProgram(const string& code) {
    istringstream ss(code);
    string line;
    while (getline(ss, line)) runLine(line);
}

#include <vector>

int main() {
    // continue: 짝수만 출력 (홀수 건너뜀)
    cout << "짝수: ";
    for (int i = 1; i <= 10; i++) {
        if (i % 2 != 0) continue;
        cout << i << " ";
    }
    cout << "\n";  // 2 4 6 8 10

    // break: 첫 번째 음수 위치 찾기
    vector<int> data = {3, 7, -2, 5, -8};
    int foundIdx = -1;
    for (int i = 0; i < (int)data.size(); i++) {
        if (data[i] < 0) { foundIdx = i; break; }
    }
    cout << "첫 음수 인덱스: " << foundIdx << "\n";  // 2

    // 중첩 루프 탈출 — bool 플래그
    bool found = false;
    int  fx = -1, fy = -1;
    int matrix[3][3] = {{1,2,3},{4,9,6},{7,8,5}};
    for (int r = 0; r < 3 && !found; r++) {
        for (int c = 0; c < 3 && !found; c++) {
            if (matrix[r][c] == 9) {
                fx = r; fy = c; found = true;
            }
        }
    }
    cout << "9의 위치: (" << fx << "," << fy << ")\n";  // (1,1)

        bool isPrime = true;
        for (int d = 2; d * d <= n; d++) {
            if (n % d == 0) { isPrime = false; break; }
        }
        if (!isPrime) continue;
        cout << n << " ";
    }
    cout << "\n";
}

#include <iostream>
using namespace std;

int main() {
    int N = 5;

    // 1. 직각삼각형
    cout << "--- 직각삼각형 ---\n";
    for (int r = 1; r <= N; r++) {
        for (int c = 0; c < r; c++) cout << "* ";
        cout << "\n";
    }

    // 2. 역삼각형
    cout << "--- 역삼각형 ---\n";
    for (int r = N; r >= 1; r--) {
        for (int c = 0; c < r; c++) cout << "* ";
        cout << "\n";
    }

    // 3. 정삼각형 (가운데 정렬)
    cout << "--- 정삼각형 ---\n";
    for (int r = 1; r <= N; r++) {
        for (int s = 0; s < N - r; s++) cout << " ";
        for (int c = 0; c < 2*r - 1; c++) cout << "*";
        cout << "\n";
    }

    // 4. 다이아몬드

void bubbleSort(vector<int>& v) {
    int n = (int)v.size();
    for (int i = 0; i < n-1; i++) {
        bool swapped = false;
        for (int j = 0; j < n-1-i; j++) {
            if (v[j] > v[j+1]) {
                int tmp = v[j]; v[j] = v[j+1]; v[j+1] = tmp;
                swapped = true;
            }
        }
        if (!swapped) break;  // 이미 정렬됨
    }
}

void selectionSort(vector<int>& v) {
    int n = (int)v.size();
    for (int i = 0; i < n-1; i++) {
        int minIdx = i;
        for (int j = i+1; j < n; j++)
            if (v[j] < v[minIdx]) minIdx = j;
        int tmp = v[i]; v[i] = v[minIdx]; v[minIdx] = tmp;
    }
}

    vector<int> b = a;

    cout << "\n";

bool isPerfect(int n) {
    int sum = 1;
    for (int d = 2; d * d <= n; d++) {
        if (n % d == 0) {
            sum += d;
            if (d != n / d) sum += n / d;
        }
    }
    return n > 1 && sum == n;
}

void matMul(const int A[][3], const int B[][3],
             int C[][3], int n) {
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < n; ++j) {
            C[i][j] = 0;
            for (int k = 0; k < n; ++k)
                C[i][j] += A[i][k] * B[k][j];
        }
}

#include <algorithm>

int main() {

// split — 구분자 기반 분리
vector<string> split(const string& s, char delim) {
    vector<string> result;
    stringstream ss(s);
    string token;
    while (getline(ss, token, delim))
        result.push_back(token);
    return result;
}

// join — 구분자로 결합
string join(const vector<string>& v, const string& sep) {
    string result;
    for (int i = 0; i < (int)v.size(); ++i) {
        if (i) result += sep;
        result += v[i];
    }
    return result;
}

// trim — 앞뒤 공백 제거
string trim(const string& s) {
    const string ws = " \t\n\r";
    size_t start = s.find_first_not_of(ws);
    if (start == string::npos) return "";
    size_t end = s.find_last_not_of(ws);
    return s.substr(start, end - start + 1);
}

// 문자 분류 카운트
void charStats(const string& s) {
    int alpha = 0, digit = 0, space = 0, other = 0;
    for (char c : s) {
        if      (isalpha(c)) ++alpha;
        else if (isdigit(c)) ++digit;
        else if (isspace(c)) ++space;
        else                 ++other;
    }
    cout << "알파=" << alpha << " 숫자=" << digit
         << " 공백=" << space << " 기타=" << other << "\n";
}

#include <algorithm>
#include <numeric>
using namespace std;

    // min_element / max_element
    cout << *min_element(v.begin(),v.end()) << "\n"; // 1
    cout << *max_element(v.begin(),v.end()) << "\n"; // 9

    // count / count_if

    cout << count_if(v.begin(),v.end(),
             [](int x){return x>4;}) << "\n";         // 3

        cout << "found 9 at idx "
             << it - v.begin() << "\n";             // 5

    sort(v.begin(),v.end());

#include <iostream>
#include <stdexcept>
using namespace std;

const int MAX = 100;
int  stk[MAX];
int  top_idx = -1;

void push(int v) {
    if (top_idx + 1 >= MAX) throw overflow_error("stack full");
    stk[++top_idx] = v;
}
int pop() {
    if (top_idx < 0) throw underflow_error("stack empty");
    return stk[top_idx--];
}
int  top()   { return stk[top_idx]; }
bool empty() { return top_idx < 0; }

// 괄호 유효성 검사 — 스택 활용
bool isValid(const string& s) {
    top_idx = -1;
    char stk2[100]; int top2 = -1;
    for (char c : s) {
        if (c=='('||c=='['||c=='{') stk2[++top2] = c;
        else if (c==')'||c==']'||c=='}') {
            if (top2<0) return false;
            char open = stk2[top2--];
            if((c==')'&&open!='(')||(c==']'&&open!='[')||(c=='}'&&open!='{'))
                return false;
        }
    }
    return top2 < 0;
}

int main() {
    push(10); push(20); push(30);
    while (!empty()) cout << pop() << " "; // 30 20 10
    cout << "\n";

    cout << isValid("{[()]}") << "\n";  // 1
    cout << isValid("{[(]}") << "\n";   // 0

#include <iostream>
using namespace std;

const int CAP = 8;
int  q[CAP];
int  head = 0, tail = 0, sz = 0;

void enqueue(int v) {
    if (sz == CAP) { cout << "full\n"; return; }
    q[tail] = v;
    tail = (tail + 1) % CAP;
    ++sz;
}
int dequeue() {
    if (!sz) { cout << "empty\n"; return -1; }
    int v = q[head];
    head = (head + 1) % CAP;
    --sz;
    return v;
}

int main() {
    for (int i = 1; i <= 5; ++i) enqueue(i * 10);
    while (sz) cout << dequeue() << " "; // 10 20 30 40 50
    cout << "\n";
    return 0;
}

#include <vector>
#include <algorithm>
#include <string>
using namespace std;

struct Student {
    string name;
    int score;
    int grade;
};

int main() {
    vector<Student> students = {
        {"Charlie", 85, 3},
        {"Alice",   92, 2},
        {"Bob",     85, 1},
        {"Diana",   78, 2},
        {"Eve",     92, 3},
    };

    // 점수 내림차순
    sort(students.begin(), students.end(),
         [](const Student& a, const Student& b) {
             return a.score > b.score;
         });
    cout << "점수 내림차순:\n";
    for (auto& s : students)
        cout << s.name << " " << s.score << "\n";

    // 다중 키: 점수 내림차순 → 이름 오름차순
    stable_sort(students.begin(), students.end(),
        [](const Student& a, const Student& b) {
            if (a.score != b.score) return a.score > b.score;
            return a.name < b.name;

        cout << s.name << " " << s.score << "\n";

    partial_sort(v.begin(), v.begin()+3, v.end());

// 슬라이딩 윈도우 최댓값 — deque에 인덱스 저장
vector<int> slideMax(const vector<int>& a, int k) {
    deque<int> dq;  // 인덱스 저장 (단조 감소)
    vector<int> result;
    for (int i = 0; i < (int)a.size(); ++i) {
        while (!dq.empty() && a[dq.back()] <= a[i])
            dq.pop_back();
        dq.push_back(i);
        if (dq.front() < i - k + 1) dq.pop_front();
        if (i >= k - 1) result.push_back(a[dq.front()]);
    }
    return result;
}

int main() {
    // deque 기본 사용
    deque<int> dq;
    dq.push_back(2);
    dq.push_back(3);
    dq.push_front(1);  // {1,2,3}
    dq.push_front(0);  // {0,1,2,3}
    dq.pop_front();    // {1,2,3}
    dq.pop_back();     // {1,2}
    for (int x : dq) cout << x << " "; // 1 2
    cout << "\n";

    // 슬라이딩 윈도우 k=3
    vector<int> a = {1,3,-1,-3,5,3,6,7};
    auto res = slideMax(a, 3);
    for (int x : res) cout << x << " "; // 3 3 5 5 6 7
    cout << "\n";
    return 0;
}

값 전달(by value) : 인수를 복사. 함수 내 변경이 원본에 영향 없음. 소형 타입(int, double)에 적합.

참조 전달(by reference, &) : 원본을 직접 접근. 복사 없이 수정 가능. C++만의 기능.

포인터 전달(by pointer, *) : 주소를 전달. nullptr 가능. C 스타일 코드와 호환. 역참조(*) 필요.

선택 기준 : 수정 필요 없으면 값 전달(소형) 또는 const 참조(대형). 수정 필요하면 참조 전달. null 가능성 있으면 포인터.

#include <iostream>
using namespace std;

// ① 값 전달 — 원본 변경 불가
void byValue(int x) {
    x = 999; // 복사본만 변경, 원본 무관
}

// ② 참조 전달 — 원본 직접 수정
void byRef(int& x) {
    x = 999; // 원본 변경됨
}

// ③ 포인터 전달 — 역참조(*) 로 원본 수정
void byPtr(int* p) {
    if (p) *p = 999; // null 체크 후 수정
}

// ④ swap 구현 비교
void swapRef(int& a, int& b) {
    int tmp = a; a = b; b = tmp;
}
void swapPtr(int* a, int* b) {
    int tmp = *a; *a = *b; *b = tmp;
}

int main() {
    int n = 10;

    byValue(n);
    cout << "byValue: " << n << "\n"; // 10 (불변)

    byRef(n);
    cout << "byRef:   " << n << "\n"; // 999

    n = 10;
    byPtr(&n);
    cout << "byPtr:   " << n << "\n"; // 999

    // swap
    int x = 1, y = 2;
    swapRef(x, y);
    cout << "swapRef: x=" << x << " y=" << y << "\n"; // 2 1

    swapPtr(&x, &y);
    cout << "swapPtr: x=" << x << " y=" << y << "\n"; // 1 2

    return 0;
}

#include <iostream>

using namespace std;

// 출력 매개변수로 두 값 반환 (이차방정식 근)
// 반환값: 근의 개수(0/1/2), r1·r2에 근 저장
int quadratic(double a, double b, double c,
               double& r1, double& r2) {
    double d = b*b - 4*a*c;
    if (d < 0) return 0;
    if (d == 0) { r1 = r2 = -b / (2*a); return 1; }
    r1 = (-b + sqrt(d)) / (2*a);
    r2 = (-b - sqrt(d)) / (2*a);
    return 2;
}

// 통계: 평균·표준편차 동시 계산
void stats(const double* data, int n,
            double& mean, double& stddev) {
    mean = 0;
    for (int i = 0; i < n; ++i) mean += data[i];
    mean /= n;
    stddev = 0;
    for (int i = 0; i < n; ++i)
        stddev += (data[i]-mean)*(data[i]-mean);
    stddev = sqrt(stddev / n);
}

int main() {
    double r1, r2;
    int cnt = quadratic(1, -5, 6, r1, r2); // x²-5x+6=0
    cout << "근의 수: " << cnt << "\n";
    cout << "r1=" << r1 << " r2=" << r2 << "\n"; // 3, 2

    double d[] = {2, 4, 4, 4, 5, 5, 7, 9};
    double m, s;
    stats(d, 8, m, s);
    cout << "평균=" << m << " 표준편차=" << s << "\n";
    return 0;

// 정렬된 배열에서 target 인덱스 반환 (없으면 -1)
int binarySearch(const int* arr, int lo, int hi, int target) {
    if (lo > hi) return -1;           // 기저 사례: 범위 소진
    int mid = lo + (hi - lo) / 2;
    if (arr[mid] == target) return mid;
    if (arr[mid] <  target) return binarySearch(arr, mid+1, hi, target);
    return binarySearch(arr, lo, mid-1, target);
}

#include <iostream>
using namespace std;

int main() {
    int a = 10, b = 20;

    // [=] 값 캡처: 호출 시점의 a,b 복사본 사용
    auto byVal = [=]() { cout << "a=" << a << " b=" << b << "\n"; };
    a = 99;               // 캡처 후 변경 — 람다에는 영향 없음

    // [x, &y] 혼합 캡처
    int x = 1, y = 2;
    auto mixed = [x, &y]() {
        cout << "x=" << x << " y=" << y << "\n";
    };
    x = 100; y = 200;
    mixed();              // x=1 (복사) y=200 (참조)

    // mutable: 값 캡처 변수 수정 허용
    int cnt = 0;
    auto counter = [cnt]() mutable {
        ++cnt;            // 복사본 수정
        cout << "내부 cnt=" << cnt << "\n";
    };
    counter(); counter(); counter();  // 1, 2, 3
    cout << "원본 cnt=" << cnt << "\n"; // 0 (원본 불변)
    return 0;
}

pair<int,int> minMaxIdx(const vector<int>& v) {

    for (int i = 1; i < (int)v.size(); ++i) {

#include <iostream>
#include <algorithm>
#include <functional>
using namespace std;

bool ascend(int a, int b)  { return a < b; }
bool descend(int a, int b) { return a > b; }

void bubbleSort(int* arr, int n, bool(*cmp)(int,int)) {
    for (int i = 0; i < n-1; ++i)
        for (int j = 0; j < n-i-1; ++j)
            if (cmp(arr[j+1], arr[j]))
                swap(arr[j], arr[j+1]);
}

int main() {
    int a[] = {5, 2, 8, 1, 9, 3};
    int b[] = {5, 2, 8, 1, 9, 3};
    int n = 6;

    bubbleSort(a, n, ascend);
    for (int x : a) cout << x << " "; // 1 2 3 5 8 9
    cout << "\n";

    bubbleSort(b, n, descend);
    for (int x : b) cout << x << " "; // 9 8 5 3 2 1
    cout << "\n";

    // 람다로 같은 효과
    vector<int> v = {5, 2, 8, 1, 9};

    for (int x : v) cout << x << " "; // 9 8 5 2 1
    cout << "\n";
    return 0;
}

포인터 vs 참조자
#include <iostream>
using namespace std;

// 포인터로 swap
void swapPtr(int* a, int* b) {
    int tmp = *a; *a = *b; *b = tmp;
}

// 참조자로 swap (더 깔끔)
void swapRef(int& a, int& b) {
    int tmp = a; a = b; b = tmp;
}

int main() {
    int x = 10;

    // 포인터
    int* p = &x;          // x의 주소를 p에 저장
    cout << *p << "\n";  // 10 (역참조)
    *p = 20;             // x도 20으로 변경
    cout << x << "\n";  // 20

    // 참조자
    int& ref = x;         // x의 별명
    ref = 30;             // x도 30으로 변경
    cout << x << "\n";  // 30

    int a = 1, b = 2;
    swapRef(a, b);
    cout << a << " " << b << "\n";  // 2 1

    return 0;
}

#include <iostream>
#include <stdexcept>
using namespace std;

class IntArray {
private:
    int* data;
    int  size_;

public:
    explicit IntArray(int n) : size_(n) {
        data = new int[n]();         // 동적 할당 (0 초기화)
        cout << "IntArray(" << n << ") 생성\n";
    }

    ~IntArray() {
        delete[] data;               // 자동 해제 — 절대 누락 없음
        cout << "IntArray 소멸\n";
    }

    int& operator[](int i) {
        if (i < 0 || i >= size_)
            throw out_of_range("index out of range");
        return data[i];
    }
    int size() const { return size_; }
};

void useArray() {
    IntArray arr(5);           // 생성
    for (int i = 0; i < arr.size(); ++i) arr[i] = (i+1) * 10;
    for (int i = 0; i < arr.size(); ++i) cout << arr[i] << " ";
    cout << "\n";
}                               // 함수 종료 시 arr 소멸 → delete[] 자동 호출

int main() {
    useArray();
    cout << "main 계속\n";
    return 0;
}

#include <string>
#include <vector>
using namespace std;

class Matrix {
private:
    vector<vector<double>> data;
    int rows, cols;
    mutable double cachedDet = 0;     // mutable: const 함수에서도 수정
    mutable bool   detValid  = false;

public:
    Matrix(int r, int c)
        : rows(r), cols(c),
          data(r, vector<double>(c, 0)) {}

    // 비const: 쓰기 가능
    double& operator()(int r, int c)       { detValid=false; return data[r][c]; }
    // const: 읽기 전용
    const double& operator()(int r, int c) const { return data[r][c]; }

    int getRows() const { return rows; }
    int getCols() const { return cols; }

    // 2×2 행렬식 — 결과 캐싱
    double det2x2() const {
        if (detValid) { cout << "[캐시 사용]\n"; return cachedDet; }
        cachedDet = data[0][0]*data[1][1] - data[0][1]*data[1][0];
        detValid  = true;
        cout << "[계산]\n";
        return cachedDet;
    }
};

int main() {
    Matrix m(2, 2);
    m(0,0)=1; m(0,1)=2;
    m(1,0)=3; m(1,1)=4;
    cout << m.det2x2() << "\n";  // [계산] -2
    cout << m.det2x2() << "\n";  // [캐시 사용] -2

    const Matrix& cm = m;
    cout << cm(0,0) << "\n";     // 1 (const 버전 호출)
    return 0;
}

#include <iostream>
#include <cmath>
using namespace std;

class Vec2 {
public:
    double x, y;
    Vec2(double x=0, double y=0) : x(x), y(y) {}

    // 이항: 멤버 함수
    Vec2 operator+(const Vec2& o) const { return {x+o.x, y+o.y}; }
    Vec2 operator-(const Vec2& o) const { return {x-o.x, y-o.y}; }
    Vec2 operator*(double s)        const { return {x*s, y*s}; }

    // 복합 대입
    Vec2& operator+=(const Vec2& o) { x+=o.x; y+=o.y; return *this; }

    // 단항 부정
    Vec2 operator-() const { return {-x, -y}; }

    // 비교
    bool operator==(const Vec2& o) const { return x==o.x && y==o.y; }
    bool operator!=(const Vec2& o) const { return !(*this==o); }

    double dot(const Vec2& o)   const { return x*o.x + y*o.y; }
    double length()              const { return sqrt(x*x + y*y); }

    // 스트림 — friend 전역 함수
    friend ostream& operator<<(ostream& os, const Vec2& v) {
        return os << "(" << v.x << "," << v.y << ")";
    }
    friend istream& operator>>(istream& is, Vec2& v) {
        return is >> v.x >> v.y;
    }
    // 스칼라 * Vec2 (비대칭 — 전역 friend)
    friend Vec2 operator*(double s, const Vec2& v) {
        return {s*v.x, s*v.y};
    }
};

int main() {
    Vec2 a(1, 2), b(3, 4);
    cout << a + b << "\n";        // (4,6)
    cout << a * 2 << "\n";       // (2,4)
    cout << 3 * a << "\n";       // (3,6)
    cout << a.dot(b) << "\n";   // 11
    cout << (a == b) << "\n";   // 0
    return 0;
}

#include <iostream>
#include <stdexcept>
using namespace std;

class Counter {
    int val;
public:
    explicit Counter(int v=0) : val(v) {}
    int get() const { return val; }

    Counter& operator++()    { ++val; return *this; }       // 전위
    Counter  operator++(int) { Counter tmp=*this; ++val; return tmp; } // 후위
    Counter& operator--()    { --val; return *this; }
    Counter  operator--(int) { Counter tmp=*this; --val; return tmp; }

    friend ostream& operator<<(ostream& os, const Counter& c) {
        return os << c.val;
    }
};

int main() {
    Counter c(5);
    cout << c++ << "\n";  // 5 (후위: 증가 전 값)
    cout << c   << "\n";  // 6
    cout << ++c << "\n";  // 7 (전위: 증가 후 값)
    return 0;
}

복사 생성자 · 이동 생성자 · Rule of Five

얕은 복사 vs 깊은 복사 : 포인터 멤버를 가진 클래스는 컴파일러 기본 복사(얕은)로는 이중 해제 발생. 깊은 복사 필요.

Rule of Three : 소멸자, 복사 생성자, 복사 대입 연산자 — 하나 필요하면 셋 다 정의.

Rule of Five(C++11) : Rule of Three + 이동 생성자 + 이동 대입 연산자.

이동(move) 의미론 : 복사 대신 자원 소유권 이전. 임시 객체(rvalue)에서 불필요한 복사 제거.

Rule of Five — DynArray
#include <iostream>
#include <algorithm>
#include <utility>   // std::move
using namespace std;

class DynArray {
private:
    int* data;
    int  size_;

public:
    // 1. 생성자
    explicit DynArray(int n, int val=0)
        : size_(n), data(new int[n]) {
        fill(data, data+n, val);
        cout << "생성("<<n<<")\n";
    }

    // 2. 소멸자
    ~DynArray() { delete[] data; cout << "소멸\n"; }

    // 3. 복사 생성자 — 깊은 복사
    DynArray(const DynArray& o)
        : size_(o.size_), data(new int[o.size_]) {
        copy(o.data, o.data+size_, data);
        cout << "복사 생성\n";
    }

    // 4. 복사 대입 — copy-and-swap
    DynArray& operator=(DynArray o) {   // 값 전달로 복사
        swap(data,   o.data);
        swap(size_,  o.size_);
        cout << "복사 대입\n";
        return *this;
    }

    // 5. 이동 생성자 — 소유권 이전
    DynArray(DynArray&& o) noexcept
        : size_(o.size_), data(o.data) {
        o.data  = nullptr;  // 원본 무효화
        o.size_ = 0;
        cout << "이동 생성\n";
    }

    // 6. 이동 대입
    DynArray& operator=(DynArray&& o) noexcept {
        if (this != &o) {
            delete[] data;
            data = o.data; size_ = o.size_;
            o.data = nullptr; o.size_ = 0;
        }
        cout << "이동 대입\n";
        return *this;
    }

    int size() const { return size_; }
    int& operator[](int i) { return data[i]; }
};

int main() {
    DynArray a(3, 1);
    DynArray b = a;              // 복사 생성
    DynArray c = move(a);       // 이동 생성
    cout << "c.size=" << c.size() << "\n"; // 3
    return 0;
}

#include <iostream>
#include <stdexcept>
using namespace std;

template<typename T, int CAP = 64>
class Stack {
private:
    T   data[CAP];
    int top_ = -1;

public:
    void push(const T& v) {
        if (top_+1 >= CAP) throw overflow_error("full");
        data[++top_] = v;
    }
    T pop() {
        if (top_ < 0) throw underflow_error("empty");
        return data[top_--];
    }
    const T& top()   const { return data[top_]; }
    bool     empty() const { return top_ < 0; }
    int      size()  const { return top_ + 1; }
};

template<typename A, typename B>
class Pair {
public:
    A first; B second;
    Pair(const A& a, const B& b) : first(a), second(b) {}
    friend ostream& operator<<(ostream& os, const Pair& p) {
        return os << "(" << p.first << "," << p.second << ")";
    }
};

int main() {
    Stack<int> si;
    si.push(10); si.push(20); si.push(30);
    while (!si.empty()) cout << si.pop() << " "; // 30 20 10
    cout << "\n";

    Stack<string, 8> ss;   // 용량 8짜리 string 스택
    ss.push("hello"); ss.push("world");
    cout << ss.top() << "\n"; // world

    Pair<string,int> p("나이", 25);

#include <iostream>
#include <vector>
#include <memory>
#include <string>
#include <algorithm>
#include <numeric>
using namespace std;

class Account {
protected:
    string owner;
    double balance;
    static int totalAccounts;

public:
    Account(const string& o, double b)
        : owner(o), balance(b) { ++totalAccounts; }
    virtual ~Account() { --totalAccounts; }

    virtual void deposit(double amt)   = 0;
    virtual bool withdraw(double amt)  = 0;

    double getBalance() const { return balance; }
    static int getCount() { return totalAccounts; }

    friend ostream& operator<<(ostream& os, const Account& a) {
        return os << "계좌[" << a.owner << "]: " << a.balance << "원";
    }
};
int Account::totalAccounts = 0;

class SavingsAccount : public Account {
    double annualRate;
public:
    SavingsAccount(const string& o, double b, double rate)
        : Account(o, b), annualRate(rate) {}

    void deposit(double amt) override {
        if (amt > 0) balance += amt;
    }
    bool withdraw(double amt) override {
        if (amt > balance) return false;
        balance -= amt; return true;
    }
    void applyInterest() {
        balance *= (1 + annualRate/12);
    }
};

class CheckingAccount : public Account {
    double feeRate;
public:
    CheckingAccount(const string& o, double b, double fee)
        : Account(o, b), feeRate(fee) {}

    void deposit(double amt) override {
        if (amt > 0) balance += amt;
    }
    bool withdraw(double amt) override {
        double total = amt * (1 + feeRate);
        if (total > balance) return false;
        balance -= total; return true;
    }
};

int main() {
    vector<unique_ptr<Account>> accounts;
    accounts.push_back(make_unique<SavingsAccount>("홍길동",10000,0.06));
    accounts.push_back(make_unique<CheckingAccount>("김철수",10000,0.01));

    static_cast<SavingsAccount*>(accounts[0].get())->applyInterest();
    accounts[1]->withdraw(1000); // 수수료 1% 포함 차감

    double total = 0;
    for (const auto& a : accounts) {
        cout << *a << "\n";
        total += a->getBalance();
    }
    cout << "총 잔액: " << total << "원\n";
    cout << "계좌 수: " << Account::getCount() << "\n";
    return 0;
}

struct Record {
    string name;
    int age;
    double score;
};

void writeCSV(const string& filename, const vector<Record>& records) {
    ofstream out(filename);
    if (!out) throw runtime_error("파일 열기 실패: " + filename);
    out << "name,age,score\n";
    for (auto& r : records)
        out << r.name << "," << r.age << "," << r.score << "\n";
}

vector<Record> readCSV(const string& filename) {
    ifstream in(filename);
    if (!in) throw runtime_error("파일 열기 실패");
    vector<Record> result;
    string line;
    getline(in, line);  // 헤더 스킵
    while (getline(in, line)) {
        istringstream ss(line);
        Record r;
        string token;
        getline(ss, r.name, ',');
        getline(ss, token, ','); r.age = stoi(token);
        getline(ss, token);    r.score = stod(token);
        result.push_back(r);
    }
    return result;
}

int main() {
    vector<Record> data = {
        {"Alice", 20, 95.5}, {"Bob", 22, 88.0}, {"Carol", 21, 76.3}
    };
    writeCSV("students.csv", data);
    cout << "CSV 저장 완료\n";

    auto loaded = readCSV("students.csv");
    cout << "=== 불러온 데이터 ===\n";
    for (auto& r : loaded)
        cout << r.name << " age=" << r.age << " score=" << r.score << "\n";

이동 생성자 / 이동 대입 연산자
#include <iostream>
#include <utility>
using namespace std;

class Buffer {
    int*  data_;
    int   size_;
public:
    // 생성자
    Buffer(int n) : size_(n), data_(new int[n]{}) {
        cout << "생성 size=" << n << "\n";
    }
    // 소멸자
    ~Buffer() { delete[] data_; cout << "소멸\n"; }

    // 복사 생성자
    Buffer(const Buffer& o) : size_(o.size_), data_(new int[o.size_]) {
        copy(o.data_, o.data_+size_, data_);
        cout << "복사\n";
    }
    // 이동 생성자
    Buffer(Buffer&& o) noexcept
        : size_(o.size_), data_(o.data_) {
        o.data_ = nullptr; o.size_ = 0;
        cout << "이동\n";
    }
    // 복사 대입
    Buffer& operator=(const Buffer& o) {
        if (this != &o) {
            delete[] data_;
            size_ = o.size_; data_ = new int[size_];
            copy(o.data_, o.data_+size_, data_);
        }
        cout << "복사대입\n"; return *this;
    }
    // 이동 대입
    Buffer& operator=(Buffer&& o) noexcept {
        if (this != &o) {
            delete[] data_;
            data_ = o.data_; size_ = o.size_;
            o.data_ = nullptr; o.size_ = 0;
        }
        cout << "이동대입\n"; return *this;
    }
    int size() const { return size_; }
};

bool isPrime(int n) {

        if (n % i == 0) return false;
    return true;

vector<int> sieve(int limit) {
    vector<bool> isP(limit+1, true);
    isP[0] = isP[1] = false;
    for (int i = 2; i*i <= limit; i++)
        if (isP[i])
            for (int j = i*i; j <= limit; j += i)
                isP[j] = false;
    vector<int> primes;
    for (int i = 2; i <= limit; i++)
        if (isP[i]) primes.push_back(i);
    return primes;
}

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

// 이진 탐색 (직접 구현)
int binarySearch(const vector<int>& v, int target) {
    int lo = 0, hi = (int)v.size() - 1;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        if (v[mid] == target) return mid;
        if (v[mid] < target) lo = mid + 1;
        else hi = mid - 1;
    }
    return -1;
}

// 2×2 행렬 곱셈
using Matrix = vector<vector<int>>;

Matrix matMul(const Matrix& A, const Matrix& B) {
    int n = (int)A.size();
    Matrix C(n, vector<int>(n, 0));
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            for (int k = 0; k < n; k++)
                C[i][j] += A[i][k] * B[k][j];
    return C;
}

int main() {
    vector<int> v = {2,5,8,12,16,23,38,56,72,91};
    cout << "23의 인덱스: " << binarySearch(v, 23) << "\n";
    cout << "99의 인덱스: " << binarySearch(v, 99) << "\n";

    Matrix B = {{5,6},{7,8}};
    Matrix C = matMul(A, B);

#include <iostream>
using namespace std;

template<typename T>
class LinkedList {
    struct Node {
        T data;
        Node* next;
        Node(T v) : data(v), next(nullptr) {}
    };
    Node* head_ = nullptr;
    int   size_ = 0;
public:
    ~LinkedList() { while (head_) popFront(); }

    void pushFront(T v) {
        Node* n = new Node(v);
        n->next = head_; head_ = n; size_++;
    }
    void pushBack(T v) {
        if (!head_) { pushFront(v); return; }
        Node* cur = head_;
        while (cur->next) cur = cur->next;
        cur->next = new Node(v); size_++;
    }
    void popFront() {
        if (!head_) return;
        Node* tmp = head_; head_ = head_->next;
        delete tmp; size_--;
    }
    void print() const {
        for (Node* c = head_; c; c = c->next)
            cout << c->data << " → ";
        cout << "null\n";
    }
    void reverse() {
        Node *prev=nullptr, *cur=head_, *next=nullptr;
        while (cur) {
            next = cur->next;
            cur->next = prev;
            prev = cur; cur = next;
        }
        head_ = prev;
    }
    int size() const { return size_; }
};

int main() {
    LinkedList<int> list;
    for (int i = 1; i <= 5; i++) list.pushBack(i);
    list.print();
    list.reverse();
    list.print();
    cout << "size=" << list.size() << "\n";
}
